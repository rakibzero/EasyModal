<task id="T4" type="execute">
<action>
Create `packages/shared/src/types.ts` exporting the four core types (multi-account-ready schema from PRODUCTION-PLAN.md §5):

```typescript
export interface Account {
  id: string;
  label: string;
  modalTokenId: string;
  /** Encrypted at rest — never plaintext on disk */
  modalTokenSecret: string;
  createdAt: string;
}

export type InstanceStatus =
  | 'idle'
  | 'building'
  | 'downloading'
  | 'serving'
  | 'ready'
  | 'failed'
  | 'cold';

export interface DeployConfig {
  gpu: string;          // e.g. 'A100-80GB'
  maxInputs: number;    // default 2 (avoids OOM)
  timeoutSeconds: number;
  appName: string;
}

export interface Instance {
  id: string;
  accountId: string;
  name: string;
  status: InstanceStatus;
  modalUrl?: string;
  config: DeployConfig;
  lastDeployedAt?: string;
}

export type LogLevel = 'info' | 'warn' | 'error' | 'success';
export type Milestone =
  | 'image-building'
  | 'models-downloading'
  | 'comfyui-starting'
  | 'url-ready'
  | 'failed';

export interface LogEvent {
  timestamp: string;    // ISO 8601
  level: LogLevel;
  message: string;
  milestone?: Milestone;
  instanceId?: string;
}
```

Create `packages/shared/src/index.ts` that re-exports everything from `./types.js` (note: with NodeNext/Bundler ESM, use `.js` extension in imports or rely on bundler resolution). Ensure `packages/shared/tsconfig.json` outputs to `dist/`.
